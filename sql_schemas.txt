-- ۱. جدول کاربران و پرسنل (سازگار با Directus BaaS)
CREATE TABLE IF NOT EXISTS users_and_staff (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(100) UNIQUE NOT NULL,
    full_name VARCHAR(200) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    department VARCHAR(150) NOT NULL,
    role VARCHAR(100) NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    external_ids JSONB DEFAULT '{}'::jsonb,
    attachments JSONB DEFAULT '[]'::jsonb, -- پیوست‌های سراسری (مجموعه فایل‌های Directus)
    
    -- فیلدهای عمومی حسابرسی و فیلد متادیتای منعطف

    date_created TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    date_updated TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    user_created UUID NULL,
    user_updated UUID NULL,
    is_deleted BOOLEAN DEFAULT FALSE

-- ۴. جدول واسط حاضرین در صورتجلسه (جلوگیری از ساختار آرایه‌ای و نرمال‌سازی روابط Many-to-Many)

    meeting_id UUID REFERENCES meeting_minutes(id) ON DELETE CASCADE,

COMMENT ON COLUMN meeting_attendees.user_id IS 'شناسه پرسنل حاضر در جلسه | Label: شناسه حاضر';

-- ۸. جدول دستور کار تفکیکی - تسک‌اوردر (تغییر قیدهای حذف از SET NULL به RESTRICT جهت ممانعت از ایجاد تسک‌های بی‌سرپرست - انطباق با سناریو ۹.۲)
CREATE TABLE IF NOT EXISTS task_orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    to_number VARCHAR(100) UNIQUE NOT NULL DEFAULT ('TO-' || nextval('task_orders_seq')),
    title VARCHAR(255) NOT NULL,
    inspector_id UUID REFERENCES users_and_staff(id) ON DELETE SET NULL NULL,
    attachments JSONB DEFAULT '[]'::jsonb, -- نقشه‌های کارگاهی، دیتیل‌های جوش و...
    
    -- تغییر مهم: قید حذف RESTRICT برای ممانعت فیزیکی دیتابیس از حذف اسناد والد در صورت وجود تسک کارگاهی فعال

    parent_meeting_item_id UUID REFERENCES meeting_items(id) ON DELETE RESTRICT NULL,
    parent_emergency_id UUID REFERENCES emergency_requests(id) ON DELETE RESTRICT NULL,
    
    -- واحد مهندسی
    engineering_requirements TEXT NULL,
    engineering_documents TEXT NULL,
    mto_reference_number VARCHAR(100) NULL,
    required_materials TEXT NULL,
    estimated_execution_time VARCHAR(150) NULL,
    
    -- واحد ایمنی
    hse_requirements TEXT NULL,
    
    -- واحد اجرا
    execution_description_and_obstacles TEXT NULL,
